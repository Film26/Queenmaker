} function updateDashboard(data) { if (!data || data.length === 0) return; lastLoadedData = data; const isYearly = currentFilterType === 'All Years'; // Data structures for aggregation const periods = {}; // key -> { orders: 0, revenue: 0, customers: Set } const customerFirstDate = {}; // id -> { val, date } const customerSubFirstDate = {}; // id_sub -> { val, date } const subChannels = {}; // subName -> { revenue: 0, buyers: Set, newCustomers: 0, newToSub: 0 } // First pass: Track overall and sub-channel first purchase dates data.forEach(row => { const id = row['Customer ID'] || row['รหัสลูกค้า (ลูกค้า) ไม่ใช้'] || row['Phone']; const sub = (row['ช่องทาง'] || row['ชื่อคลัง (คลังสินค้า)'] || 'Unknown').toString().trim().toUpperCase(); const dateStr = row['วันที่สร้าง'] || row['วันที่โอนเงิน']; const revenueStr = row['ยอดขาย'] || row['ราคาสินค้ายังไม่รวมภาษี']; if (sub === 'MAIN') return; if (!id || !dateStr) return; const date = parseDate(dateStr); if (!date) return; const revenue = parseFloat(revenueStr?.toString().replace(/,/g, '') || 0); // Track overall first purchase if (!customerFirstDate[id] || date.val < customerFirstDate[id].val) { customerFirstDate[id] = { val: date.val, date: date }; } // Track first purchase in this sub-channel const subKey = `${id}_${sub}`; if (!customerSubFirstDate[subKey] || date.val < customerSubFirstDate[subKey].val) { customerSubFirstDate[subKey] = { val: date.val, date: date }; } const pKey = isYearly ? date.year : date.month; if (!periods[pKey]) { periods[pKey] = { orders: 0, revenue: 0, customers: new Set() }; } periods[pKey].orders++; periods[pKey].revenue += revenue; periods[pKey].customers.add(id); }); // Second pass: Calculate metrics for selected period const monthsInOrder = Object.keys(periods).sort((a, b) => b - a); if (monthsInOrder.length === 0) return; const selectedMonthKey = isYearly ? monthsInOrder[0] : monthsInOrder[0]; // Take latest const monthNames = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December']; document.getElementById('strategy-period-label').textContent = isYearly ? `Year ${selectedMonthKey}` : monthNames[selectedMonthKey]; data.forEach(row => { const id = row['Customer ID'] || row['รหัสลูกค้า (ลูกค้า) ไม่ใช้'] || row['Phone']; const sub = (row['ช่องทาง'] || row['ชื่อคลัง (คลังสินค้า)'] || 'Unknown').toString().trim().toUpperCase(); const dateStr = row['วันที่สร้าง'] || row['วันที่โอนเงิน']; const revenueStr = row['ยอดขาย'] || row['ราคาสินค้ายังไม่รวมภาษี']; if (sub === 'MAIN') return; if (!id || !dateStr) return; const date = parseDate(dateStr); if (!date) return; const pKey = isYearly ? date.year : date.month; if (pKey.toString() !== selectedMonthKey.toString()) return; const revenue = parseFloat(revenueStr?.toString().replace(/,/g, '') || 0); if (!subChannels[sub]) { subChannels[sub] = { revenue: 0, buyers: new Set(), newCustomers: 0, newToSub: 0 }; } subChannels[sub].revenue += revenue; subChannels[sub].buyers.add(id); if (customerFirstDate[id].val === date.val) { subChannels[sub].newCustomers++; } const subKey = `${id}_${sub}`; if (customerSubFirstDate[subKey].val === date.val) { subChannels[sub].newToSub++; } }); // Population of Strategy Table const tableBody = document.getElementById('strategy-table-body'); if (tableBody) { tableBody.innerHTML = ''; Object.keys(subChannels).sort((a,b) => subChannels[b].revenue - subChannels[a].revenue).forEach(sub => { const s = subChannels[sub]; const buyers = s.buyers.size; if (buyers === 0) return; const migPct = ((s.newToSub / buyers) * 100).toFixed(0); const newPct = ((s.newCustomers / buyers) * 100).toFixed(0); let strategy = ''; let strategyClass = ''; if (newPct > 70) { strategy = '🔵 Vanguard (ทัพหน้า)'; strategyClass = 'strat-vanguard'; } else if (migPct > 70) { strategy = '🟢 Migration Hub'; strategyClass = 'strat-migration'; } else if (migPct > 30 || newPct > 10) { strategy = '🟠 Retention Hub'; strategyClass = 'strat-retention'; } else { strategy = '🔴 Cash Cow (เสือนอนกิน)'; strategyClass = 'strat-cashcow'; } const tr = document.createElement('tr'); tr.innerHTML = ` ${sub} ${s.revenue.toLocaleString()} ${buyers} ${s.newCustomers} ${s.newToSub} ${migPct}% ${newPct}% ${strategy} `; tableBody.appendChild(tr); }); } // Group New Customers by their first purchase period for the Overview chart const periodNewCust = {}; Object.values(customerFirstDate).forEach(d => { const pKey = isYearly ? d.date.year : d.date.month; periodNewCust[pKey] = (periodNewCust[pKey] || 0) + 1; }); // Prepare Labels and Data for Overview let labels = []; if (!isYearly) { labels = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec']; } else { labels = Object.keys(periods).sort((a, b) => a - b); } const growthData = labels.map((l, i) => { const key = isYearly ? l : i; return periods[key]?.customers.size || 0; }); const freqData = labels.map((l, i) => { const key = isYearly ? l : i; const p = periods[key]; return p ? (p.orders / p.customers.size).toFixed(2) : 0; }); const spendingData = labels.map((l, i) => { const key = isYearly ? l : i; const p = periods[key]; return p ? (p.revenue / p.customers.size).toFixed(0) : 0; }); const newCustData = labels.map((l, i) => { const key = isYearly ? l : i; return periodNewCust[key] || 0; }); // Update DOM const tc = document.getElementById('totalCustomers'); if (tc) tc.textContent = Object.keys(customerFirstDate).length.toLocaleString(); const currentMonth = new Date().getMonth(); const newThisMonth = periodNewCust[currentMonth] || 0; const nc = document.getElementById('newCustomers'); if (nc) nc.textContent = newThisMonth.toLocaleString(); // Update Charts if (charts.growth) updateChart(charts.growth, labels, growthData); if (charts.buyerAnalytics) updateChart(charts.buyerAnalytics, labels, growthData, freqData); if (charts.retention) updateChart(charts.retention, labels, spendingData, newCustData); // Update Sparklines randomly for aesthetic if (charts.spark1) updateSparkline(charts.spark1); if (charts.spark2) updateSparkline(charts.spark2); } function updateChart(chart, labels, data1, data2 = null) { if (!chart) return; chart.data.labels = labels; chart.data.datasets[0].data = data1; if (data2 && chart.data.datasets[1]) { chart.data.datasets[1].data = data2; } chart.update(); } function updateSparkline(chart) { if (!chart) return; chart.data.datasets[0].data = chart.data.datasets[0].data.map(() => Math.floor(Math.random() * 50) + 10); chart.update(); }
